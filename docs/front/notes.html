// 输入搜索框初始化
var countries = ["测试", "测试1", "测试2", "测试3"];
$('#test').autocomplete({
    // 搜索数组
    lookup: countries,
    // 选中回调事件
    onSelect: function (suggestion) {
        // suggestion为选中的对象
        console.log(suggestion);
    }
});
